// HUNTER SKILL - PHANTASMIC ARROW
//============================================================
payon_in02,54,13,7	script	Albest	55,{
	mes "[Albest]";
	if(BaseJob!=Job_Hunter) goto L_hunterno;
	if(getskilllv("HT_PHANTASMIC")) goto L_alreadyhave;
	if(JobLevel<40) goto L_nojob;

	mes "Eu irei ensinar uma técnica";
	mes "passada de geração em geração";
	mes "na minha família. Minha";
	mes "família guardou esse segredo por";
	mes "muitos anos, mas eu irei ensinar";
	mes "por alguns itens em troca.";
	next;
	mes "[Albest]";
	mes "Eu preciso só de algumas";
	mes "coisas. Aqui está a minha lista:";
	mes "5 Rubi Amaldiçoado";
	mes "5 Pena de Harpia";
	mes "30 Ração para bichinho.";
	next;
	
	mes "[Albest]";
	mes "Deixe-me checar os seus itens.";
	next;
	
	if(countitem(724)<5 || countitem(7115)<5 || countitem(537)<30) goto L_noitems;//Items: Cursed_Ruby, Harpy_Feather, Pet_Food,
	delitem 724, 5;//Items: Cursed_Ruby,
	delitem 7115, 5;//Items: Harpy_Feather,
	delitem 537, 30;//Items: Pet_Food,
	
	mes "[Albest]";
	mes "Eu estou vendo que você tem o que precisa,";
	mes "agora vou ensinar o meu talento.";
	skill "HT_PHANTASMIC",1,0;
	close;

L_noitems:
	mes "[Albest]";
	mes "Você não tem todos os itens.";
	mes "Volte até aqui quando você tiver";
	mes "todos os itens que eu pedi.";
	close;
		
L_nojob:
	mes "Volte aqui quando você aprender";
	mes "mais sobre ser um Caçador.";
	mes "Você precisa de pelo menos 40";
	mes "níveis de classe para se tornar um Caçador.";
	close;

L_alreadyhave:
	mes "Ah, meu aprendiz, você";
	mes "veio para me visitar...";
	close;
	
L_hunterno:
	mes "Existe algo em que eu";
	mes "possa ajudá-lo? Eu acho que";
	mes "você me confundiu com outro velho.";
	close;
}
